# Coupling Matrix Eigenvalue Computation

import numpy as np

# L₄-derived constants
phi = (1 + np.sqrt(5)) / 2          # φ = 1.618...
tau = 1 / phi                        # τ = φ⁻¹ = 0.618...
Delta = tau**4                       # Δ = φ⁻⁴ = 0.1459...

z_c = np.sqrt(3) / 2                 # z_c = √3/2 = 0.866...
L4 = 7                               # Lucas anchor

# Threshold values (z-coordinates)
z = np.array([
    tau,                              # α: Paradox
    1 - Delta,                        # β: Activation
    z_c,                              # γ: Lens (critical)
    np.sqrt(2) - 0.5,                 # δ: Ignition
    K                                 # ε: K-Formation
])

# Compatibility matrix C (from BFADGS composition)
C = np.array([
    [0,    tau,    tau,    tau,    tau   ],   # α
    [tau,  0,      1.0,    tau,    tau**2],   # β
    [tau,  1.0,    0,      tau,    K     ],   # γ
    [tau,  tau,    tau,    0,      1.0   ],   # δ
    [tau,  tau**2, K,      1.0,    0     ]    # ε
])

def cascade(z_bar, sigma_sq=0.003):
    """Resonance enhancement near Lens threshold."""
    return 1 + 0.5 * np.exp(-(z_bar - z_c)**2 / sigma_sq)

eigenvectors = eigenvectors[:, idx]
